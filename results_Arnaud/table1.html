Table by IMD

table_imd <- dataset %>%
  group_by(imd) %>%
  summarise(
    n_patients = n(),
    n_with_medication = sum(has_medication == 1, na.rm = TRUE),
    percent_with_medication =
      round(100 * n_with_medication / n_patients, 1),
    .groups = "drop"
  ) %>%
  arrange(imd)

Patients over time by IMD

table_imd_plot <- dataset %>%
  group_by(index_date, imd) %>%
  summarise(
    n_patients = n(),
    .groups = "drop"
  ) %>%
  filter(imd != "Missing")

ggplot(table_imd_plot,aes(x = index_date,y = n_patients,colour = imd, group = imd)) +
  geom_line(linewidth = 0.6) +
  geom_point(size = 1.8) +
  labs(
    x = "Month",
    y = "Number of patients",
    colour = "IMD",
    title = "Figure 1: Number of patients receiving medication overtime by IMD"
  ) +
  scale_x_date(
    date_breaks = "1 months",
    date_labels = "%Y-%m"
  ) +
  theme_bw() +
  theme(
    axis.text.x = element_text(
      angle = 45,
      hjust = 1
    ),
    legend.position = "bottom"
  )
